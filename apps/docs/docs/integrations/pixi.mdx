## 安装 / Install

```bash
pnpm add @tiny-playworks/game-ui-pixi @tiny-playworks/game-ui-runtime pixi.js
```

## 用法 / Usage

```ts
import { createGameUiRuntime } from '@tiny-playworks/game-ui-runtime';
import { createPixiFeedbackBridge } from '@tiny-playworks/game-ui-pixi';

const runtime = createGameUiRuntime();

const detach = createPixiFeedbackBridge(runtime, {
  container: feedbackLayer,
  factory: {
    createLabel(text, options) {
      const label = new PIXI.Text({ text, style: { fill: options?.variant === 'heal' ? '#5cff9d' : '#ff4d7d' } });
      return label;
    },
  },
  stageWidth: 1280,
  stageHeight: 720,
});

runtime.emitDamage({ value: 128, variant: 'critical', anchor: { x: 50, y: 40 } });
```
